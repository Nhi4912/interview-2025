```javascript
class SymbolTable {
  constructor(parent = null) {
    this.parent = parent;
    this.symbols = new Map();
  }

  define(name, type, value) {
    if (this.symbols.has(name)) {
      throw new Error(`Symbol '${name}' already defined in this scope`);
    }
    
    this.symbols.set(name, { type, value, defined: true });
  }

  lookup(name) {
    if (this.symbols.has(name)) {
      return this.symbols.get(name);
    }
    
    if (this.parent) {
      return this.parent.lookup(name);
    }
    
    return null;
  }

  update(name, value) {
    if (this.symbols.has(name)) {
      const symbol = this.symbols.get(name);
      symbol.value = value;
      return true;
    }
    
    if (this.parent) {
      return this.parent.update(name, value);
    }
    
    return false;
  }

  enterScope() {
    return new SymbolTable(this);
  }

  exitScope() {
    return this.parent;
  }
}
```

```javascript
class TypeChecker extends ASTVisitor {
  constructor() {
    super();
    this.symbolTable = new SymbolTable();
    this.errors = [];
  }

  error(message, node) {
    this.errors.push({
      message,
      line: node.line,
      column: node.column
    });
  }

  visitVariableDeclaration(node) {
    node.declarations.forEach(decl => {
      const name = decl.id.name;
      
      let type = 'any';
      if (decl.init) {
        type = this.visit(decl.init);
      }
      
      try {
        this.symbolTable.define(name, type);
      } catch (e) {
        this.error(e.message, decl.id);
      }
    });
  }

  visitFunctionDeclaration(node) {
    const name = node.id.name;
    const paramTypes = node.params.map(() => 'any');
    
    this.symbolTable.define(name, {
      kind: 'function',
      params: paramTypes,
      returnType: 'any'
    });
    
    // Enter function scope
    this.symbolTable = this.symbolTable.enterScope();
    
    // Define parameters
    node.params.forEach((param, i) => {
      this.symbolTable.define(param.name, paramTypes[i]);
    });
    
    // Check body
    this.visit(node.body);
    
    // Exit function scope
    this.symbolTable = this.symbolTable.exitScope();
  }

  visitBinaryExpression(node) {
    const leftType = this.visit(node.left);
    const rightType = this.visit(node.right);
    
    if (node.operator === '+') {
      if (leftType === 'string' || rightType === 'string') {
        return 'string';
      }
      if (leftType === 'number' && rightType === 'number') {
        return 'number';
      }
      this.error(`Cannot apply '+' to types ${leftType} and ${rightType}`, node);
      return 'any';
    }
    
    if (['-', '*', '/'].includes(node.operator)) {
      if (leftType !== 'number' || rightType !== 'number') {
        this.error(`Operator '${node.operator}' requires number operands`, node);
      }
      return 'number';
    }
    
    if (['==', '!=', '<', '>', '<=', '>='].includes(node.operator)) {
      return 'boolean';
    }
    
    return 'any';
  }

  visitIdentifier(node) {
    const symbol = this.symbolTable.lookup(node.name);
    
    if (!symbol) {
      this.error(`Undefined variable '${node.name}'`, node);
      return 'any';
    }
    
    return symbol.type;
  }

  visitLiteral(node) {
    if (typeof node.value === 'number') return 'number';
    if (typeof node.value === 'string') return 'string';
    if (typeof node.value === 'boolean') return 'boolean';
    if (node.value === null) return 'null';
    return 'any';
  }

  visitCallExpression(node) {
    const calleeType = this.visit(node.callee);
    
    if (calleeType.kind !== 'function') {
      this.error('Cannot call non-function', node);
      return 'any';
    }
    
    // Check argument count
    if (node.arguments.length !== calleeType.params.length) {
      this.error(
        `Expected ${calleeType.params.length} arguments, got ${node.arguments.length}`,
        node
      );
    }
    
    // Check argument types
    node.arguments.forEach((arg, i) => {
      const argType = this.visit(arg);
      const paramType = calleeType.params[i];
      
      if (paramType !== 'any' && argType !== paramType) {
        this.error(
          `Argument ${i + 1}: expected ${paramType}, got ${argType}`,
          arg
        );
      }
    });
    
    return calleeType.returnType;
  }
}
```

```javascript
class CodeGenerator extends ASTVisitor {
  constructor() {
    super();
    this.code = '';
    this.indentLevel = 0;
  }

  indent() {
    return '  '.repeat(this.indentLevel);
  }

  emit(code) {
    this.code += code;
  }

  emitLine(code = '') {
    this.code += this.indent() + code + '\n';
  }

  visitProgram(node) {
    node.body.forEach(stmt => this.visit(stmt));
    return this.code;
  }

  visitVariableDeclaration(node) {
    node.declarations.forEach(decl => {
      this.emit(this.indent() + node.kind + ' ');
      this.visit(decl.id);
      
      if (decl.init) {
        this.emit(' = ');
        this.visit(decl.init);
      }
      
      this.emit(';\n');
    });
  }

  visitFunctionDeclaration(node) {
    this.emit(this.indent() + 'function ');
    this.visit(node.id);
    this.emit('(');
    
    node.params.forEach((param, i) => {
      if (i > 0) this.emit(', ');
      this.visit(param);
    });
    
    this.emit(') ');
    this.visit(node.body);
    this.emit('\n');
  }

  visitBlockStatement(node) {
    this.emit('{\n');
    this.indentLevel++;
    
    node.body.forEach(stmt => this.visit(stmt));
    
    this.indentLevel--;
    this.emit(this.indent() + '}');
  }

  visitReturnStatement(node) {
    this.emit(this.indent() + 'return');
    
    if (node.argument) {
      this.emit(' ');
      this.visit(node.argument);
    }
    
    this.emit(';\n');
  }

  visitIfStatement(node) {
    this.emit(this.indent() + 'if (');
    this.visit(node.test);
    this.emit(') ');
    
    if (node.consequent.type === 'BlockStatement') {
      this.visit(node.consequent);
    } else {
      this.emit('\n');
      this.indentLevel++;
      this.visit(node.consequent);
      this.indentLevel--;
    }
    
    if (node.alternate) {
      this.emit(' else ');
      
      if (node.alternate.type === 'BlockStatement' || node.alternate.type === 'IfStatement') {
        this.visit(node.alternate);
      } else {
        this.emit('\n');
        this.indentLevel++;
        this.visit(node.alternate);
        this.indentLevel--;
      }
    }
    
    this.emit('\n');
  }

  visitWhileStatement(node) {
    this.emit(this.indent() + 'while (');
    this.visit(node.test);
    this.emit(') ');
    this.visit(node.body);
    this.emit('\n');
  }

  visitExpressionStatement(node) {
    this.emit(this.indent());
    this.visit(node.expression);
    this.emit(';\n');
  }

  visitBinaryExpression(node) {
    this.visit(node.left);
    this.emit(' ' + node.operator + ' ');
    this.visit(node.right);
  }

  visitAssignmentExpression(node) {
    this.visit(node.left);
    this.emit(' ' + node.operator + ' ');
    this.visit(node.right);
  }

  visitCallExpression(node) {
    this.visit(node.callee);
    this.emit('(');
    
    node.arguments.forEach((arg, i) => {
      if (i > 0) this.emit(', ');
      this.visit(arg);
    });
    
    this.emit(')');
  }

  visitMemberExpression(node) {
    this.visit(node.object);
    
    if (node.computed) {
      this.emit('[');
      this.visit(node.property);
      this.emit(']');
    } else {
      this.emit('.');
      this.visit(node.property);
    }
  }

  visitIdentifier(node) {
    this.emit(node.name);
  }

  visitLiteral(node) {
    if (typeof node.value === 'string') {
      this.emit('"' + node.value + '"');
    } else {
      this.emit(String(node.value));
    }
  }
}
```

```javascript
class ConstantFolder extends ASTVisitor {
  visitBinaryExpression(node) {
    // Recursively optimize children
    node.left = this.visit(node.left);
    node.right = this.visit(node.right);
    
    // If both operands are literals, compute at compile time
    if (node.left.type === 'Literal' && node.right.type === 'Literal') {
      const left = node.left.value;
      const right = node.right.value;
      let result;
      
      switch (node.operator) {
        case '+': result = left + right; break;
        case '-': result = left - right; break;
        case '*': result = left * right; break;
        case '/': result = left / right; break;
        case '==': result = left == right; break;
        case '!=': result = left != right; break;
        case '<': result = left < right; break;
        case '>': result = left > right; break;
        case '<=': result = left <= right; break;
        case '>=': result = left >= right; break;
        default: return node;
      }
      
      return { type: 'Literal', value: result };
    }
    
    return node;
  }

  visitUnaryExpression(node) {
    node.argument = this.visit(node.argument);
    
    if (node.argument.type === 'Literal') {
      const value = node.argument.value;
      let result;
      
      switch (node.operator) {
        case '+': result = +value; break;
        case '-': result = -value; break;
        case '!': result = !value; break;
        default: return node;
      }
      
      return { type: 'Literal', value: result };
    }
    
    return node;
  }
}
```

```javascript
class DeadCodeEliminator extends ASTVisitor {
  visitIfStatement(node) {
    node.test = this.visit(node.test);
    
    // If test is a constant
    if (node.test.type === 'Literal') {
      if (node.test.value) {
        // Condition is always true, return consequent
        return this.visit(node.consequent);
      } else {
        // Condition is always false, return alternate or nothing
        return node.alternate ? this.visit(node.alternate) : null;
      }
    }
    
    node.consequent = this.visit(node.consequent);
    if (node.alternate) {
      node.alternate = this.visit(node.alternate);
    }
    
    return node;
  }

  visitBlockStatement(node) {
    const newBody = [];
    let reachable = true;
    
    for (const stmt of node.body) {
      if (!reachable) {
        // Skip unreachable code
        continue;
      }
      
      const visited = this.visit(stmt);
      if (visited) {
        newBody.push(visited);
      }
      
      // Check if this statement makes subsequent code unreachable
      if (stmt.type === 'ReturnStatement' || 
          stmt.type === 'ThrowStatement' ||
          stmt.type === 'BreakStatement' ||
          stmt.type === 'ContinueStatement') {
        reachable = false;
      }
    }
    
    node.body = newBody;
    return node;
  }
}
```

```javascript
class FunctionInliner extends ASTVisitor {
  constructor(maxSize = 10) {
    super();
    this.maxSize = maxSize;
    this.functions = new Map();
  }

  visitFunctionDeclaration(node) {
    // Store function for potential inlining
    const size = this.estimateSize(node.body);
    
    if (size <= this.maxSize) {
      this.functions.set(node.id.name, {
        params: node.params,
        body: node.body
      });
    }
    
    return node;
  }

  visitCallExpression(node) {
    // Check if this is a call to an inlineable function
    if (node.callee.type === 'Identifier') {
      const func = this.functions.get(node.callee.name);
      
      if (func && node.arguments.length === func.params.length) {
        // Create inline block
        const inlineBlock = {
          type: 'BlockStatement',
          body: []
        };
        
        // Add parameter assignments
        func.params.forEach((param, i) => {
          inlineBlock.body.push({
            type: 'VariableDeclaration',
            kind: 'const',
            declarations: [{
              type: 'VariableDeclarator',
              id: param,
              init: node.arguments[i]
            }]
          });
        });
        
        // Add function body
        inlineBlock.body.push(...func.body.body);
        
        return inlineBlock;
      }
    }
    
    return node;
  }

  estimateSize(node) {
    let size = 1;
    
    for (const key in node) {
      const value = node[key];
      
      if (Array.isArray(value)) {
        value.forEach(item => {
          if (item && typeof item === 'object' && item.type) {
            size += this.estimateSize(item);
          }
        });
      } else if (value && typeof value === 'object' && value.type) {
        size += this.estimateSize(value);
      }
    }
    
    return size;
  }
}
```

```javascript
module.exports = function(babel) {
  const { types: t } = babel;
  
  return {
    name: 'my-plugin',
    visitor: {
      // Transform arrow functions to regular functions
      ArrowFunctionExpression(path) {
        const { node } = path;
        
        // Create function expression
        const func = t.functionExpression(
          null,
          node.params,
          t.blockStatement(
            node.body.type === 'BlockStatement'
              ? node.body.body
              : [t.returnStatement(node.body)]
          )
        );
        
        // Replace arrow function with regular function
        path.replaceWith(func);
      },
      
      // Transform template literals
      TemplateLiteral(path) {
        const { node } = path;
        
        if (node.expressions.length === 0) {
          // Simple template literal with no expressions
          path.replaceWith(t.stringLiteral(node.quasis[0].value.cooked));
        } else {
          // Template literal with expressions
          let result = t.stringLiteral(node.quasis[0].value.cooked);
          
          for (let i = 0; i < node.expressions.length; i++) {
            result = t.binaryExpression(
              '+',
              result,
              node.expressions[i]
            );
            
            if (i < node.quasis.length - 1) {
              result = t.binaryExpression(
                '+',
                result,
                t.stringLiteral(node.quasis[i + 1].value.cooked)
              );
            }
          }
          
          path.replaceWith(result);
        }
      },
      
      // Transform class declarations
      ClassDeclaration(path) {
        const { node } = path;
        const className = node.id.name;
        
        // Create constructor function
        const constructor = node.body.body.find(
          method => method.kind === 'constructor'
        );
        
        const constructorFunc = t.functionDeclaration(
          t.identifier(className),
          constructor ? constructor.params : [],
          constructor ? constructor.body : t.blockStatement([])
        );
        
        // Create prototype methods
        const methods = node.body.body
          .filter(method => method.kind === 'method')
          .map(method => {
            return t.expressionStatement(
              t.assignmentExpression(
                '=',
                t.memberExpression(
                  t.memberExpression(
                    t.identifier(className),
                    t.identifier('prototype')
                  ),
                  t.identifier(method.key.name)
                ),
                t.functionExpression(
                  null,
                  method.params,
                  method.body
                )
              )
            );
          });
        
        // Replace class with constructor and methods
        path.replaceWithMultiple([constructorFunc, ...methods]);
      }
    }
  };
};
```

```javascript
// TypeScript type checker concepts
class TypeScriptTypeChecker {
  constructor() {
    this.types = new Map();
    this.constraints = [];
  }

  // Structural type checking
  isAssignable(source, target) {
    // Primitive types
    if (source === target) return true;
    
    // Any type
    if (target === 'any') return true;
    if (source === 'never') return true;
    
    // Union types
    if (target.kind === 'union') {
      return target.types.some(t => this.isAssignable(source, t));
    }
    
    if (source.kind === 'union') {
      return source.types.every(t => this.isAssignable(t, target));
    }
    
    // Object types (structural)
    if (source.kind === 'object' && target.kind === 'object') {
      // Check if source has all properties of target
      for (const [key, targetProp] of Object.entries(target.properties)) {
        const sourceProp = source.properties[key];
        
        if (!sourceProp) {
          if (!targetProp.optional) return false;
        } else {
          if (!this.isAssignable(sourceProp.type, targetProp.type)) {
            return false;
          }
        }
      }
      
      return true;
    }
    
    // Function types
    if (source.kind === 'function' && target.kind === 'function') {
      // Parameters are contravariant
      if (source.parameters.length < target.parameters.length) {
        return false;
      }
      
      for (let i = 0; i < target.parameters.length; i++) {
        if (!this.isAssignable(target.parameters[i], source.parameters[i])) {
          return false;
        }
      }
      
      // Return type is covariant
      return this.isAssignable(source.returnType, target.returnType);
    }
    
    return false;
  }

  // Type inference
  infer(node) {
    switch (node.type) {
      case 'Literal':
        return this.inferLiteral(node);
      
      case 'BinaryExpression':
        return this.inferBinaryExpression(node);
      
      case 'CallExpression':
        return this.inferCallExpression(node);
      
      case 'ArrayExpression':
        return this.inferArrayExpression(node);
      
      case 'ObjectExpression':
        return this.inferObjectExpression(node);
      
      default:
        return 'any';
    }
  }

  inferLiteral(node) {
    const value = node.value;
    
    if (typeof value === 'number') return 'number';
    if (typeof value === 'string') return 'string';
    if (typeof value === 'boolean') return 'boolean';
    if (value === null) return 'null';
    if (value === undefined) return 'undefined';
    
    return 'any';
  }

  inferBinaryExpression(node) {
    const leftType = this.infer(node.left);
    const rightType = this.infer(node.right);
    
    if (node.operator === '+') {
      if (leftType === 'string' || rightType === 'string') {
        return 'string';
      }
      return 'number';
    }
    
    if (['-', '*', '/', '%'].includes(node.operator)) {
      return 'number';
    }
    
    if (['==', '!=', '<', '>', '<=', '>=', '&&', '||'].includes(node.operator)) {
      return 'boolean';
    }
    
    return 'any';
  }

  inferCallExpression(node) {
    const calleeType = this.infer(node.callee);
    
    if (calleeType.kind === 'function') {
      return calleeType.returnType;
    }
    
    return 'any';
  }

  inferArrayExpression(node) {
    if (node.elements.length === 0) {
      return { kind: 'array', elementType: 'any' };
    }
    
    // Infer union of element types
    const elementTypes = node.elements.map(el => this.infer(el));
    const uniqueTypes = [...new Set(elementTypes)];
    
    if (uniqueTypes.length === 1) {
      return { kind: 'array', elementType: uniqueTypes[0] };
    }
    
    return {
      kind: 'array',
      elementType: { kind: 'union', types: uniqueTypes }
    };
  }

  inferObjectExpression(node) {
    const properties = {};
    
    node.properties.forEach(prop => {
      const key = prop.key.name || prop.key.value;
      properties[key] = {
        type: this.infer(prop.value),
        optional: false
      };
    });
    
    return { kind: 'object', properties };
  }
}
```

```javascript
// Simplified webpack module bundler
class ModuleBundler {
  constructor(options) {
    this.entry = options.entry;
    this.output = options.output;
    this.modules = new Map();
    this.moduleId = 0;
  }

  async build() {
    // Start from entry point
    await this.loadModule(this.entry);
    
    // Generate bundle
    return this.generateBundle();
  }

  async loadModule(filepath) {
    if (this.modules.has(filepath)) {
      return this.modules.get(filepath);
    }
    
    // Read file
    const content = await this.readFile(filepath);
    
    // Parse to find dependencies
    const dependencies = this.findDependencies(content);
    
    // Create module
    const module = {
      id: this.moduleId++,
      filepath,
      content,
      dependencies
    };
    
    this.modules.set(filepath, module);
    
    // Recursively load dependencies
    for (const dep of dependencies) {
      const depPath = this.resolvePath(filepath, dep);
      await this.loadModule(depPath);
    }
    
    return module;
  }

  findDependencies(content) {
    const dependencies = [];
    
    // Find require() calls
    const requireRegex = /require\(['"]([^'"]+)['"]\)/g;
    let match;
    
    while ((match = requireRegex.exec(content)) !== null) {
      dependencies.push(match[1]);
    }
    
    // Find import statements
    const importRegex = /import .+ from ['"]([^'"]+)['"]/g;
    
    while ((match = importRegex.exec(content)) !== null) {
      dependencies.push(match[1]);
    }
    
    return dependencies;
  }

  generateBundle() {
    const modules = Array.from(this.modules.values());
    
    // Create module map
    const moduleMap = modules.map(mod => {
      return `${mod.id}: function(module, exports, require) {
        ${mod.content}
      }`;
    }).join(',\n');
    
    // Generate bundle code
    return `
(function(modules) {
  const installedModules = {};
  
  function require(moduleId) {
    if (installedModules[moduleId]) {
      return installedModules[moduleId].exports;
    }
    
    const module = installedModules[moduleId] = {
      id: moduleId,
      exports: {}
    };
    
    modules[moduleId](module, module.exports, require);
    
    return module.exports;
  }
  
  return require(0);
})({
  ${moduleMap}
});
    `;
  }

  resolvePath(from, to) {
    // Simplified path resolution
    if (to.startsWith('./') || to.startsWith('../')) {
      return path.resolve(path.dirname(from), to);
    }
    return to;
  }

  async readFile(filepath) {
    // Simplified file reading
    return fs.promises.readFile(filepath, 'utf-8');
  }
}
```

### Tree Shaking

```javascript
class TreeShaker {
  constructor(ast) {
    this.ast = ast;
    this.usedExports = new Set();
    this.exportedNames = new Map();
  }

  shake() {
    // Phase 1: Find all exports
    this.findExports();
    
    // Phase 2: Mark used exports
    this.markUsedExports();
    
    // Phase 3: Remove unused exports
    return this.removeUnusedExports();
  }

  findExports() {
    this.ast.body.forEach(node => {
      if (node.type === 'ExportNamedDeclaration') {
        if (node.declaration) {
          if (node.declaration.type === 'VariableDeclaration') {
            node.declaration.declarations.forEach(decl => {
              this.exportedNames.set(decl.id.name, node);
            });
          } else if (node.declaration.id) {
            this.exportedNames.set(node.declaration.id.name, node);
          }
        }
      }
    });
  }

  markUsedExports() {
    // Find all import statements
    this.ast.body.forEach(node => {
      if (node.type === 'ImportDeclaration') {
        node.specifiers.forEach(spec => {
          if (spec.type === 'ImportSpecifier') {
            this.usedExports.add(spec.imported.name);
          }
        });
      }
    });
    
    // Find all references in code
    const visitor = new ASTVisitor();
    visitor.visitIdentifier = (node) => {
      if (this.exportedNames.has(node.name)) {
        this.usedExports.add(node.name);
      }
    };
    visitor.visit(this.ast);
  }

  removeUnusedExports() {
    const newBody = this.ast.body.filter(node => {
      if (node.type === 'ExportNamedDeclaration') {
        if (node.declaration) {
          if (node.declaration.type === 'VariableDeclaration') {
            node.declaration.declarations = node.declaration.declarations.filter(
              decl => this.usedExports.has(decl.id.name)
            );
            return node.declaration.declarations.length > 0;
          } else if (node.declaration.id) {
            return this.usedExports.has(node.declaration.id.name);
          }
        }
      }
      return true;
    });
    
    return { ...this.ast, body: newBody };
  }
}
```
